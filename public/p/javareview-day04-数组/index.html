     public int halfSearch (int[] arr, int key){
       int min = 0;
       int max = arr.length - 1;
       int mid = (min + max)/2;
       while(arr[mid] != key){
         if(key < arr[mid]){
           max = mid - 1;
         }else if(key > arr[mid]){
           min = mid + 1;
         }
         mid = (min + max)/2;
       }
       return mid;
     }

            int temp = num & 15;
